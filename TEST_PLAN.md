```python
async def test_delete_single_scan(page):
    """Удаление единичного скана"""
    await page.goto("/history")
    await page.click("button[aria-label='Удалить']")
    await page.click("button:has-text('Удалить')")
    await page.wait_for_selector("text=Отчет успешно удален")

async def test_clear_all_scans(page):
    """Массовое удаление"""
    await page.goto("/history")
    await page.click("button:has-text('Очистить историю')")
    await page.click("button:has-text('Удалить все')")
    await page.wait_for_selector("text=Вся история успешно удалена")
```
